import plotly.graph_objects as go
import pandas as pd

# Let's assume 'df' is your DataFrame after reading the CSV
df = pd.read_csv('data/Dheeraj_data/national_arrests.csv')

# Create traces for each offense
traces = []
for offense in df.columns[1:]:  # Assuming the first column is 'data_year'
    traces.append(go.Scatter(
        x=df['data_year'],
        y=df[offense],
        mode='lines',
        name=offense
    ))

# Create the figure with all the traces
fig = go.Figure(traces)

# Update layout with titles and axes labels
fig.update_layout(
    title='All Arrest by Offense Year-Over-Year',
    xaxis_title='Year',
    yaxis_title='Number of Arrests',
    hovermode='x unified'  # Groups hover info at the vertical level
)

# Show the plot
fig.show()
import plotly.graph_objects as go
import pandas as pd


df = pd.read_csv('data/Dheeraj_data/national_arrests.csv')
# Create traces for each offense
traces = []
for offense in df.columns[1:]:  # Assuming the first column is 'data_year'
    traces.append(go.Scatter(
        x=df['data_year'],
        y=df[offense],
        mode='lines+markers',  # Use both lines and markers
        name=offense,
        line=dict(width=2),  # Increase line width for visibility
        marker=dict(size=4),  # Adjust marker size as needed
        text=offense,  # Set the hover text to offense name
        hoverinfo='y+text'  # Show only the y-value and offense name on hover
    ))

# Create the figure with all the traces
fig = go.Figure(traces)

# Update layout with titles and axes labels
fig.update_layout(
    title='All Arrest by Offense Year-Over-Year',
    xaxis_title='Year',
    yaxis_title='Number of Arrests',
    hovermode='closest',  # Show hover info for the closest point
    legend=dict(  # Adjust the legend's appearance
        title=dict(text='Offenses'),
        orientation='h',  # Horizontal layout to save vertical space
        yanchor='bottom',
        y=1.02,  # Position the legend above the plot
        xanchor='right',
        x=1
    ),
    width=800,  # Wider plot
    height=600  # Taller plot
)

fig.add_vline(x=2020, line_width=2, line_dash="dash", line_color="red")

# Add annotation to the vertical line
fig.add_annotation(
    x=2020, y=max(df.max()),  # Adjust the y position according to your data
    text="Start of COVID-19 impact",
    showarrow=True,
    arrowhead=1,
    ax=-50,
    ay=0
)

# Add interactive hover functionality to toggle traces on and off
fig.update_traces(
    hoverlabel=dict(namelength=-1)  # Show full name of the offense on hover
)

# Show the plot
fig.show()
